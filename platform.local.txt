# Allow importing files directly from a config folder in the sketch directory.
# This allow per project configuration (for projects that are aware of this,
# that is). Used mostly to setting defines during compilation.
#
# Note that this seem to only work if added as extra flags. Adding them
# directly to flags in platform.txt did not work.

compiler.c.extra_flags=-I{build.path}/sketch/config
compiler.cpp.extra_flags=-I{build.path}/sketch/config
